## 💡 优化建议（可选）

### 1. 日志文件持久化（可选）

如果需要将日志保存到文件系统，可以在 `logger.js` 中添加：

```javascript
// 保存日志到文件
async saveLogsToFile() {
  if (!pathManager.isWPSAvailable()) return
  
  const logsDir = pathManager.getLogsDir()
  if (!logsDir) return
  
  try {
    // 确保日志目录存在
    pathManager.ensureDir(logsDir)
    
    const fs = window.Application.FileSystem
    const date = new Date().toISOString().slice(0, 10)
    const logFile = `${logsDir}/app-${date}.log`
    
    const content = this.exportLogsAsText()
    fs.WriteFile(logFile, content)
    
    console.log(`日志已保存到: ${logFile}`)
  } catch (error) {
    console.error('保存日志文件失败:', error)
  }
}

// 在关键时刻调用（如程序退出、错误发生等）
window.addEventListener('beforeunload', () => {
  logger.saveLogsToFile()
})
```

### 2. 数据备份功能（可选）

可以为 localStorage 数据添加文件系统备份：

```javascript
// 备份 localStorage 数据到文件
async backupLocalStorageToFile() {
  if (!pathManager.isWPSAvailable()) return
  
  const configDir = pathManager.getConfigDir()
  if (!configDir) return
  
  try {
    const fs = window.Application.FileSystem
    const backupFile = `${configDir}/localStorage-backup.json`
    
    // 导出所有 localStorage 数据
    const data = {}
    for (let i = 0; i < localStorage.length; i++) {
      const key = localStorage.key(i)
      data[key] = localStorage.getItem(key)
    }
    
    fs.WriteFile(backupFile, JSON.stringify(data, null, 2))
    console.log(`localStorage 已备份到: ${backupFile}`)
  } catch (error) {
    console.error('备份 localStorage 失败:', error)
  }
}
```
